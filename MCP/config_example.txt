# Developer Intelligence MCP Server Configuration
# Set these as environment variables before running the server

# GitHub API Configuration (optional)
# Get your token from: https://github.com/settings/tokens
# Required scopes: repo, read:user
export GITHUB_TOKEN="your_github_personal_access_token_here"
export GITHUB_USERNAME="your_github_username_here"

# Weather API Configuration (optional)
# Get your API key from: https://openweathermap.org/api
# Free tier available
export WEATHER_API_KEY="your_openweathermap_api_key_here"
export WEATHER_CITY="New York"
